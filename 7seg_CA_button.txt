char array[] = {0xC0, 0xF9, 0xA4, 0xB0, 0x99, 0x92, 0x82, 0xF8, 0x80, 0x90};
int i = 0; // Initialize outside the loop

void main(){
      TRISB = 0X00;
      TRISC = 0XFF;
      PORTB = 0XFF;
      i = 0;
      while(1){
         if(PORTC.F0 == 1){
          PORTB = array[i];
          delay_ms(200);
          i++;

          if(i > 9) i = 0;

          while(PORTC = 1)
          delay_ms(50);

         }
      }
}
